# DB 스키마 (Supabase / Postgres)

MVP 테이블: `profiles` · `facilities` · `visits` · `reviews` · `wishes`.
설계 핵심: **visits(방문일지=그날 기록)와 reviews(시설평=상시 평가) 분리**(사우나이키타이가 못 한 UX 차별점) · PostGIS 반경검색 · UGC 스펙은 `specs jsonb` · **RLS 필수**(클라이언트가 anon key로 DB 직접 접근).

Supabase SQL Editor에 그대로 실행:

```sql
-- ══════════════════════════════════════════════════════════
-- 사우나 (com.jeong.sauna) — MVP 스키마
-- ══════════════════════════════════════════════════════════
create extension if not exists postgis;

-- ── profiles (auth.users 연동) ─────────────────────────────
create table profiles (
  id         uuid primary key references auth.users(id) on delete cascade,
  nickname   text not null,
  avatar_url text,
  created_at timestamptz not null default now()
);

create or replace function handle_new_user() returns trigger
language plpgsql security definer set search_path = public as $$
begin
  insert into profiles (id, nickname)
  values (new.id, coalesce(new.raw_user_meta_data->>'nickname', '사우너'||substr(new.id::text,1,6)));
  return new;
end $$;
create trigger on_auth_user_created
  after insert on auth.users for each row execute function handle_new_user();

-- ── facilities (공공데이터 시드 5,652 + UGC 스펙) ───────────
create table facilities (
  id         bigint generated always as identity primary key,
  name       text not null,
  category   text not null check (category in ('목욕탕/사우나','찜질방','한증막','온천','기타')),
  sauna      boolean not null default false,          -- 발한실 보유(공공데이터)
  address    text,
  sido       text,                                     -- 시/도 (지역 필터)
  sigungu    text,                                     -- 시/군/구
  phone      text,
  lat        double precision,
  lon        double precision,
  geog       geography(Point,4326),                    -- 반경 검색용
  specs      jsonb not null default '{}',              -- UGC: 노천/가족탕/24시간/수면실/불가마/온천수질...
  source     text not null default 'public-data',
  created_at timestamptz not null default now()
);
create index facilities_geog_idx     on facilities using gist (geog);
create index facilities_category_idx on facilities (category);
create index facilities_sido_idx     on facilities (sido);

create or replace function set_geog() returns trigger language plpgsql as $$
begin
  if new.lat is not null and new.lon is not null then
    new.geog := st_setsrid(st_makepoint(new.lon, new.lat),4326)::geography;
  end if;
  return new;
end $$;
create trigger facilities_set_geog
  before insert or update of lat, lon on facilities
  for each row execute function set_geog();

-- ── visits (방문일지 = 그날 기록, 여러 건) ─────────────────
create table visits (
  id          bigint generated always as identity primary key,
  user_id     uuid not null references profiles(id) on delete cascade,
  facility_id bigint not null references facilities(id) on delete cascade,
  visited_on  date not null default current_date,
  memo        text,
  photos      text[] not null default '{}',            -- storage 경로
  created_at  timestamptz not null default now()
);
create index visits_user_idx     on visits (user_id, visited_on desc);
create index visits_facility_idx on visits (facility_id, created_at desc);

-- ── reviews (시설평 = 상시 평가, 유저×시설 1건) ────────────
create table reviews (
  id          bigint generated always as identity primary key,
  user_id     uuid not null references profiles(id) on delete cascade,
  facility_id bigint not null references facilities(id) on delete cascade,
  rating      smallint not null check (rating between 1 and 5),
  body        text,
  created_at  timestamptz not null default now(),
  updated_at  timestamptz not null default now(),
  unique (user_id, facility_id)                         -- 시설당 1리뷰(수정)
);
create index reviews_facility_idx on reviews (facility_id, created_at desc);

create or replace function touch_updated_at() returns trigger language plpgsql as $$
begin new.updated_at := now(); return new; end $$;
create trigger reviews_touch before update on reviews
  for each row execute function touch_updated_at();

-- ── wishes (찜 / 가고싶다) ─────────────────────────────────
create table wishes (
  user_id     uuid not null references profiles(id) on delete cascade,
  facility_id bigint not null references facilities(id) on delete cascade,
  created_at  timestamptz not null default now(),
  primary key (user_id, facility_id)
);
create index wishes_facility_idx on wishes (facility_id);

-- ── 내 주변 시설 (지도/검색용 RPC) ─────────────────────────
create or replace function nearby_facilities(p_lat float, p_lon float, p_radius_m int default 3000)
returns setof facilities language sql stable as $$
  select * from facilities
  where geog is not null
    and st_dwithin(geog, st_setsrid(st_makepoint(p_lon,p_lat),4326)::geography, p_radius_m)
  order by geog <-> st_setsrid(st_makepoint(p_lon,p_lat),4326)::geography
  limit 100;
$$;

-- ══════════════════════════════════════════════════════════
-- RLS (클라이언트가 anon key로 DB 직접 접근 → 필수)
-- ══════════════════════════════════════════════════════════
alter table profiles   enable row level security;
alter table facilities enable row level security;
alter table visits     enable row level security;
alter table reviews    enable row level security;
alter table wishes     enable row level security;

create policy facilities_read on facilities for select using (true);

create policy profiles_read       on profiles for select using (true);
create policy profiles_write_self on profiles for update using (auth.uid() = id);

create policy visits_read       on visits for select using (true);
create policy visits_write_self on visits for all
  using (auth.uid() = user_id) with check (auth.uid() = user_id);

create policy reviews_read       on reviews for select using (true);
create policy reviews_write_self on reviews for all
  using (auth.uid() = user_id) with check (auth.uid() = user_id);

create policy wishes_self on wishes for all
  using (auth.uid() = user_id) with check (auth.uid() = user_id);
```

## 설계 노트
- **visits vs reviews 분리**: 시설 상세엔 reviews(별점·시설평) 탭, 피드엔 visits(일지) 탭.
- **specs jsonb**: 노천/가족탕/24시간/불가마종류/온천수질 등 UGC 속성. 자주 거르는 건 나중에 컬럼 승격.
- **sido/sigungu**: 시드 import 때 주소 파싱해 채움(지역 필터).
- **wishes 프라이빗**(본인만). "가고싶다 수 랭킹"은 Phase3 카운터/뷰.
- **facilities 쓰기 정책 없음**: 시드 import·관리자 수정은 service_role 키로(RLS 우회). UGC 시설정보 수정은 Phase3 `facility_edits` 검수 테이블.
- **storage**: `visits.photos`는 경로만. 버킷(`visit-photos`)은 Supabase에서 생성·정책 설정.
