# DB 스키마 (Supabase / Postgres)

MVP 테이블: `profiles` · `facilities` · `visits` · `reviews` · `wishes`.
설계 핵심: **visits(방문일지=그날 기록)와 reviews(시설평=상시 평가) 분리**(사우나이키타이가 못 한 UX 차별점) · PostGIS 반경검색 · UGC 스펙은 `specs jsonb` · **RLS 필수**(클라이언트가 anon key로 DB 직접 접근).

Supabase SQL Editor에 그대로 실행:

```sql
-- ══════════════════════════════════════════════════════════
-- 사우나 (com.jeong.sauna) — MVP 스키마
-- ══════════════════════════════════════════════════════════
create extension if not exists postgis;

-- ── profiles (auth.users 연동) ─────────────────────────────
create table profiles (
  id         uuid primary key references auth.users(id) on delete cascade,
  nickname   text not null,
  avatar_url text,
  created_at timestamptz not null default now()
);

create or replace function handle_new_user() returns trigger
language plpgsql security definer set search_path = public as $$
begin
  insert into profiles (id, nickname, avatar_url)
  values (
    new.id,
    coalesce(new.raw_user_meta_data->>'name', new.raw_user_meta_data->>'full_name', '사우너'||substr(new.id::text,1,6)),
    coalesce(new.raw_user_meta_data->>'avatar_url', new.raw_user_meta_data->>'picture')
  );
  return new;
end $$;
create trigger on_auth_user_created
  after insert on auth.users for each row execute function handle_new_user();

-- ── facilities (공공데이터 시드 5,652 + UGC 태그) ───────────
-- 카테고리 없음(순수 태그 모델). 목욕탕·찜질방·온천도 tags 안의 값.
-- 태그는 정해진 사전(docs/태그-사전.md)에서만. specs는 태그 아닌 값(온도·요금·영업시간).
create table facilities (
  id         bigint generated always as identity primary key,
  name       text not null,
  tags       text[] not null default '{}',             -- 태그 사전 값(유형·시설·물·운영)
  address    text,
  sido       text,                                     -- 시/도 (지역 필터)
  sigungu    text,                                     -- 시/군/구
  phone      text,
  lat        double precision,
  lon        double precision,
  geog       geography(Point,4326),                    -- 반경 검색용
  specs      jsonb not null default '{}',              -- 태그 아닌 값: 온도·요금·영업시간 등
  source     text not null default 'public-data',
  created_at timestamptz not null default now()
);
create index facilities_geog_idx on facilities using gist (geog);
create index facilities_tags_idx on facilities using gin (tags);   -- 태그 포함 검색(@>)
create index facilities_sido_idx on facilities (sido);

create or replace function set_geog() returns trigger language plpgsql as $$
begin
  if new.lat is not null and new.lon is not null then
    new.geog := st_setsrid(st_makepoint(new.lon, new.lat),4326)::geography;
  end if;
  return new;
end $$;
create trigger facilities_set_geog
  before insert or update of lat, lon on facilities
  for each row execute function set_geog();

-- ── posts (커뮤니티 글: 방문기록 + 자유) ───────────────────
-- facility_id 있으면 '방문기록', 없으면 '자유' (topic은 자동 파생)
create table posts (
  id           bigint generated always as identity primary key,
  user_id      uuid not null references profiles(id) on delete cascade,
  facility_id  bigint references facilities(id) on delete set null,
  topic        text generated always as (case when facility_id is null then '자유' else '방문기록' end) stored,
  body         text not null,
  photos       text[] not null default '{}',
  like_count   int not null default 0,
  report_count int not null default 0,
  status       text not null default 'visible' check (status in ('visible','blocked','hidden')),
  created_at   timestamptz not null default now()
);
create index posts_facility_idx on posts (facility_id, created_at desc);
create index posts_topic_idx    on posts (topic, created_at desc);
create index posts_user_idx     on posts (user_id, created_at desc);

-- 좋아요
create table post_likes (
  user_id    uuid not null references profiles(id) on delete cascade,
  post_id    bigint not null references posts(id) on delete cascade,
  created_at timestamptz not null default now(),
  primary key (user_id, post_id)
);
create or replace function sync_like_count() returns trigger language plpgsql as $$
begin
  if tg_op='INSERT' then update posts set like_count=like_count+1 where id=new.post_id;
  elsif tg_op='DELETE' then update posts set like_count=like_count-1 where id=old.post_id; end if;
  return null;
end $$;
create trigger post_likes_count after insert or delete on post_likes
  for each row execute function sync_like_count();

-- 신고 + 자동차단 (스토어 UGC 정책)
create table post_reports (
  id          bigint generated always as identity primary key,
  reporter_id uuid not null references profiles(id) on delete cascade,
  post_id     bigint not null references posts(id) on delete cascade,
  reason      text,
  created_at  timestamptz not null default now(),
  unique (reporter_id, post_id)                         -- 중복 신고 방지
);
create or replace function handle_report() returns trigger language plpgsql as $$
begin
  update posts set report_count = report_count + 1 where id = new.post_id;
  update posts set status = 'blocked'
    where id = new.post_id and report_count >= 5 and status = 'visible';  -- 임계값 5(조정 가능)
  return null;
end $$;
create trigger post_reports_handle after insert on post_reports
  for each row execute function handle_report();

-- 유저 차단 (Apple UGC 심사 요구)
create table user_blocks (
  blocker_id uuid not null references profiles(id) on delete cascade,
  blocked_id uuid not null references profiles(id) on delete cascade,
  created_at timestamptz not null default now(),
  primary key (blocker_id, blocked_id)
);

-- ── reviews (시설평 = 상시 평가, 유저×시설 1건) ────────────
create table reviews (
  id          bigint generated always as identity primary key,
  user_id     uuid not null references profiles(id) on delete cascade,
  facility_id bigint not null references facilities(id) on delete cascade,
  rating      smallint not null check (rating between 1 and 5),
  body        text,
  created_at  timestamptz not null default now(),
  updated_at  timestamptz not null default now(),
  unique (user_id, facility_id)                         -- 시설당 1리뷰(수정)
);
create index reviews_facility_idx on reviews (facility_id, created_at desc);

create or replace function touch_updated_at() returns trigger language plpgsql as $$
begin new.updated_at := now(); return new; end $$;
create trigger reviews_touch before update on reviews
  for each row execute function touch_updated_at();

-- ── wishes (찜 / 가고싶다) ─────────────────────────────────
create table wishes (
  user_id     uuid not null references profiles(id) on delete cascade,
  facility_id bigint not null references facilities(id) on delete cascade,
  created_at  timestamptz not null default now(),
  primary key (user_id, facility_id)
);
create index wishes_facility_idx on wishes (facility_id);

-- ── 내 주변 시설 (지도/검색용 RPC) ─────────────────────────
create or replace function nearby_facilities(p_lat float, p_lon float, p_radius_m int default 3000)
returns setof facilities language sql stable as $$
  select * from facilities
  where geog is not null
    and st_dwithin(geog, st_setsrid(st_makepoint(p_lon,p_lat),4326)::geography, p_radius_m)
  order by geog <-> st_setsrid(st_makepoint(p_lon,p_lat),4326)::geography
  limit 100;
$$;

-- ══════════════════════════════════════════════════════════
-- RLS (클라이언트가 anon key로 DB 직접 접근 → 필수)
-- ══════════════════════════════════════════════════════════
alter table profiles     enable row level security;
alter table facilities   enable row level security;
alter table posts        enable row level security;
alter table post_likes   enable row level security;
alter table post_reports enable row level security;
alter table user_blocks  enable row level security;
alter table reviews      enable row level security;
alter table wishes       enable row level security;

create policy facilities_read on facilities for select using (true);

create policy profiles_read       on profiles for select using (true);
create policy profiles_write_self on profiles for update using (auth.uid() = id);

-- posts(글): 공개(visible) + 본인 것 읽기, 쓰기는 본인만
create policy posts_read       on posts for select
  using (status = 'visible' or auth.uid() = user_id);
create policy posts_write_self on posts for all
  using (auth.uid() = user_id) with check (auth.uid() = user_id);

-- 좋아요: 읽기 공개(카운트), 본인만 쓰기
create policy post_likes_read on post_likes for select using (true);
create policy post_likes_self on post_likes for all
  using (auth.uid() = user_id) with check (auth.uid() = user_id);

-- 신고: 본인 신고만(프라이빗), 처리는 service_role
create policy post_reports_self on post_reports for all
  using (auth.uid() = reporter_id) with check (auth.uid() = reporter_id);

-- 유저 차단: 본인 것만
create policy user_blocks_self on user_blocks for all
  using (auth.uid() = blocker_id) with check (auth.uid() = blocker_id);

create policy reviews_read       on reviews for select using (true);
create policy reviews_write_self on reviews for all
  using (auth.uid() = user_id) with check (auth.uid() = user_id);

create policy wishes_self on wishes for all
  using (auth.uid() = user_id) with check (auth.uid() = user_id);
```

## 설계 노트
- **순수 태그 모델**: 카테고리 없음. 목욕탕·찜질방·온천도 `tags`. 정해진 사전([태그-사전.md](태그-사전.md))에서만. UI가 대표 태그를 간판처럼 노출.
  - 검색: `select * from facilities where tags @> array['온천']` (GIN 인덱스로 빠름). 다중: `tags @> array['찜질방','24시간']`.
- **posts vs reviews 분리**: reviews=시설 별점 평가(상시). posts=커뮤니티 글. 시설 상세는 `[리뷰]`(reviews) + `[피드]`(그 시설 posts) 탭.
- **posts.topic 자동 파생**: `facility_id` 있으면 `방문기록`, 없으면 `자유`. 별도 topic 선택 UI 없음(글쓰기 때 업장 붙이면 방문기록). N회차 = 유저의 그 시설 방문기록 posts 수.
- **모더레이션(스토어 UGC 필수)**: `post_reports` 신고 → `report_count` 증가, **5건 이상 자동 `blocked`**(임계값 조정). 피드는 `status='visible'`만. `user_blocks`로 유저 차단(Apple 요구). 앱은 차단한 유저 글 제외 쿼리. **EULA(무관용 정책) 필요** — 스토어 심사 항목.
- **좋아요**: `post_likes` + 트리거로 `like_count` 동기화.
- **specs jsonb**: 태그로 표현 안 되는 **값** — 사우나 온도, 입장요금, 영업시간 등.
- **sido/sigungu**: 시드 import 때 주소 파싱해 채움(지역 필터).
- **wishes 프라이빗**(본인만). "가고싶다 수 랭킹"은 Phase3 카운터/뷰.
- **facilities 쓰기 정책 없음**: 시드 import·관리자 수정은 service_role 키로(RLS 우회).
- **프로필**: nickname·avatar_url 편집(설정). OAuth로 프리필, 유저 수정. 아바타는 storage `avatars` 버킷.
- **storage 버킷**: `post-photos`(글 사진), `avatars`(프로필). 경로만 DB 저장, 버킷·정책은 Supabase에서 설정.
